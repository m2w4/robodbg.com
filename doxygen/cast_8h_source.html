58class type_caster_enum_type {
59private:
60 using Underlying = typename std::underlying_type<EnumType>::type;

62public:
63 static constexpr auto name = const_name<EnumType>();

65 template <typename SrcType>
66 static handle cast(SrcType &&src, return_value_policy, handle parent) {
67 handle native_enum
68 = global_internals_native_enum_type_map_get_item(std::type_index(typeid(EnumType)));
69 if (native_enum) {
70 return native_enum(static_cast<Underlying>(src)).release();
71 }
72 return type_caster_base<EnumType>::cast(
73 std::forward<SrcType>(src),
74 // Fixes https://github.com/pybind/pybind11/pull/3643#issuecomment-1022987818:
75 return_value_policy::copy,
76 parent);

80 handle native_enum
81 = global_internals_native_enum_type_map_get_item(std::type_index(typeid(EnumType)));
82 if (native_enum) {
83 if (!isinstance(src, native_enum)) {
84 return false;
85 }
86 type_caster<Underlying> underlying_caster;
87 if (!underlying_caster.load(src.attr("value"), convert)) {
88 pybind11_fail("native_enum internal consistency failure.");
89 }

91 return true;
92 }
93 if (!pybind11_enum_) {
94 pybind11_enum_.reset(new type_caster_base<EnumType>());
95 }
96 return pybind11_enum_->load(src, convert);
97 }

224template <typename T>
225struct type_caster<T, enable_if_t<std::is_arithmetic<T>::value && !is_std_char_type<T>::value>> {
226 using _py_type_0 = conditional_t<sizeof(T) <= sizeof(long), long, long long>;
227 using _py_type_1 = conditional_t<std::is_signed<T>::value,
228 _py_type_0,
229 typename std::make_unsigned<_py_type_0>::type>;
230 using py_type = conditional_t<std::is_floating_point<T>::value, double, _py_type_1>;
231
232public:
233 bool load(handle src, bool convert) {
234 py_type py_value;
235
236 if (!src) {
237 return false;
238 }
239
240#if !defined(PYPY_VERSION)
241 auto index_check = [](PyObject *o) { return PyIndex_Check(o); };
242#else
243 // In PyPy 7.3.3, `PyIndex_Check` is implemented by calling `__index__`,
244 // while CPython only considers the existence of `nb_index`/`__index__`.
245 auto index_check = [](PyObject *o) { return hasattr(o, "__index__"); };
246#endif
247
248 if (std::is_floating_point<T>::value) {
249 if (convert || PyFloat_Check(src.ptr())) {
250 py_value = (py_type) PyFloat_AsDouble(src.ptr());
251 } else {
252 return false;
253 }
254 } else if (PyFloat_Check(src.ptr())
255 || (!convert && !PYBIND11_LONG_CHECK(src.ptr()) && !index_check(src.ptr()))) {
256 return false;
257 } else {
258 handle src_or_index = src;
259 // PyPy: 7.3.7's 3.8 does not implement PyLong_*'s __index__ calls.
260#if defined(PYPY_VERSION)
261 object index;
262 if (!PYBIND11_LONG_CHECK(src.ptr())) { // So: index_check(src.ptr())
263 index = reinterpret_steal<object>(PyNumber_Index(src.ptr()));
264 if (!index) {
265 PyErr_Clear();
266 if (!convert)
267 return false;
268 } else {
269 src_or_index = index;
270 }
271 }
272#endif
273 if (std::is_unsigned<py_type>::value) {
274 py_value = as_unsigned<py_type>(src_or_index.ptr());
275 } else { // signed integer:
276 py_value = sizeof(T) <= sizeof(long)
277 ? (py_type) PyLong_AsLong(src_or_index.ptr())
278 : (py_type) PYBIND11_LONG_AS_LONGLONG(src_or_index.ptr());
279 }
280 }
281
282 // Python API reported an error
283 bool py_err = py_value == (py_type) -1 && PyErr_Occurred();
284
285 // Check to see if the conversion is valid (integers should match exactly)
286 // Signed/unsigned checks happen elsewhere
287 if (py_err
288 || (std::is_integral<T>::value && sizeof(py_type) != sizeof(T)
289 && py_value != (py_type) (T) py_value)) {
290 PyErr_Clear();
291 if (py_err && convert && (PyNumber_Check(src.ptr()) != 0)) {
292 auto tmp = reinterpret_steal<object>(std::is_floating_point<T>::value
293 ? PyNumber_Float(src.ptr())
294 : PyNumber_Long(src.ptr()));
295 PyErr_Clear();
296 return load(tmp, false);
297 }
298 return false;
299 }
300
301 value = (T) py_value;
302 return true;
303 }

305 template <typename U = T>
306 static typename std::enable_if<std::is_floating_point<U>::value, handle>::type
307 cast(U src, return_value_policy /* policy */, handle /* parent */) {
308 return PyFloat_FromDouble((double) src);
309 }

311 template <typename U = T>
312 static typename std::enable_if<!std::is_floating_point<U>::value && std::is_signed<U>::value
313 && (sizeof(U) <= sizeof(long)),
314 handle>::type
315 cast(U src, return_value_policy /* policy */, handle /* parent */) {

321 && (sizeof(U) <= sizeof(unsigned long)),
322 handle>::type
323 cast(U src, return_value_policy /* policy */, handle /* parent */) {

329 && (sizeof(U) > sizeof(long)),
330 handle>::type
331 cast(U src, return_value_policy /* policy */, handle /* parent */) {

416template <>
417class type_caster<bool> {
418public:
419 bool load(handle src, bool convert) {
420 if (!src) {
421 return false;
422 }
423 if (src.ptr() == Py_True) {
424 value = true;
425 return true;
426 }
427 if (src.ptr() == Py_False) {
428 value = false;
429 return true;
430 }
431 if (convert || is_numpy_bool(src)) {
432 // (allow non-implicit conversion for numpy booleans), use strncmp
433 // since NumPy 1.x had an additional trailing underscore.
434
435 Py_ssize_t res = -1;
436 if (src.is_none()) {
437 res = 0; // None is implicitly converted to False
438 }
439#if defined(PYPY_VERSION)
440 // On PyPy, check that "__bool__" attr exists
441 else if (hasattr(src, PYBIND11_BOOL_ATTR)) {
442 res = PyObject_IsTrue(src.ptr());
443 }
444#else
445 // Alternate approach for CPython: this does the same as the above, but optimized
446 // using the CPython API so as to avoid an unneeded attribute lookup.
447 else if (auto *tp_as_number = Py_TYPE(src.ptr())->tp_as_number) {
448 if (PYBIND11_NB_BOOL(tp_as_number)) {
449 res = (*PYBIND11_NB_BOOL(tp_as_number))(src.ptr());
450 }
451 }
452#endif
453 if (res == 0 || res == 1) {
454 value = (res != 0);
455 return true;
456 }
457 PyErr_Clear();
458 }
459 return false;
460 }
461 static handle cast(bool src, return_value_policy /* policy */, handle /* parent */) {
462 return handle(src ? Py_True : Py_False).inc_ref();
463 }
464 PYBIND11_TYPE_CASTER(bool, const_name("bool"));
465
466private:
467 // Test if an object is a NumPy boolean (without fetching the type).
468 static inline bool is_numpy_bool(handle object) {
469 const char *type_name = Py_TYPE(object.ptr())->tp_name;
470 // Name changed to `numpy.bool` in NumPy 2, `numpy.bool_` is needed for 1.x support
471 return std::strcmp("numpy.bool", type_name) == 0
472 || std::strcmp("numpy.bool_", type_name) == 0;
473 }

476// Helper class for UTF-{8,16,32} C++ stl strings:
477template <typename StringType, bool IsView = false>
478struct string_caster {
479 using CharT = typename StringType::value_type;
480
481 // Simplify life by being able to assume standard char sizes (the standard only guarantees
482 // minimums, but Python requires exact sizes)
483 static_assert(!std::is_same<CharT, char>::value || sizeof(CharT) == 1,
484 "Unsupported char size != 1");
485#if defined(PYBIND11_HAS_U8STRING)
486 static_assert(!std::is_same<CharT, char8_t>::value || sizeof(CharT) == 1,
487 "Unsupported char8_t size != 1");
488#endif
489 static_assert(!std::is_same<CharT, char16_t>::value || sizeof(CharT) == 2,
490 "Unsupported char16_t size != 2");
491 static_assert(!std::is_same<CharT, char32_t>::value || sizeof(CharT) == 4,
492 "Unsupported char32_t size != 4");
493 // wchar_t can be either 16 bits (Windows) or 32 (everywhere else)
494 static_assert(!std::is_same<CharT, wchar_t>::value || sizeof(CharT) == 2 || sizeof(CharT) == 4,
495 "Unsupported wchar_t size != 2/4");
496 static constexpr size_t UTF_N = 8 * sizeof(CharT);
497
498 bool load(handle src, bool) {
499 handle load_src = src;
500 if (!src) {
501 return false;
502 }
503 if (!PyUnicode_Check(load_src.ptr())) {
504 return load_raw(load_src);
505 }
506
507 // For UTF-8 we avoid the need for a temporary `bytes` object by using
508 // `PyUnicode_AsUTF8AndSize`.
509 if (UTF_N == 8) {
510 Py_ssize_t size = -1;
511 const auto *buffer
512 = reinterpret_cast<const CharT *>(PyUnicode_AsUTF8AndSize(load_src.ptr(), &size));
513 if (!buffer) {
514 PyErr_Clear();
515 return false;
516 }
517 value = StringType(buffer, static_cast<size_t>(size));
518 return true;
519 }
520
521 auto utfNbytes
522 = reinterpret_steal<object>(PyUnicode_AsEncodedString(load_src.ptr(),
523 UTF_N == 8 ? "utf-8"
524 : UTF_N == 16 ? "utf-16"
525 : "utf-32",
526 nullptr));
527 if (!utfNbytes) {
528 PyErr_Clear();
529 return false;
530 }
531
532 const auto *buffer
533 = reinterpret_cast<const CharT *>(PYBIND11_BYTES_AS_STRING(utfNbytes.ptr()));
534 size_t length = (size_t) PYBIND11_BYTES_SIZE(utfNbytes.ptr()) / sizeof(CharT);
535 // Skip BOM for UTF-16/32
536 if (UTF_N > 8) {
537 buffer++;
538 length--;
539 }
540 value = StringType(buffer, length);
541
542 // If we're loading a string_view we need to keep the encoded Python object alive:
543 if (IsView) {
544 loader_life_support::add_patient(utfNbytes);
545 }
546
547 return true;
548 }
549
550 static handle
551 cast(const StringType &src, return_value_policy /* policy */, handle /* parent */) {
552 const char *buffer = reinterpret_cast<const char *>(src.data());
553 auto nbytes = ssize_t(src.size() * sizeof(CharT));
554 handle s = decode_utfN(buffer, nbytes);
555 if (!s) {
556 throw error_already_set();
557 }
558 return s;
559 }
560
561 PYBIND11_TYPE_CASTER(StringType, const_name(PYBIND11_STRING_NAME));

563private:
564 static handle decode_utfN(const char *buffer, ssize_t nbytes) {
565#if !defined(PYPY_VERSION)
566 return UTF_N == 8 ? PyUnicode_DecodeUTF8(buffer, nbytes, nullptr)
567 : UTF_N == 16 ? PyUnicode_DecodeUTF16(buffer, nbytes, nullptr, nullptr)
568 : PyUnicode_DecodeUTF32(buffer, nbytes, nullptr, nullptr);
569#else
570 // PyPy segfaults when on PyUnicode_DecodeUTF16 (and possibly on PyUnicode_DecodeUTF32 as
571 // well), so bypass the whole thing by just passing the encoding as a string value, which
572 // works properly:
573 return PyUnicode_Decode(buffer,
574 nbytes,
575 UTF_N == 8 ? "utf-8"
576 : UTF_N == 16 ? "utf-16"
577 : "utf-32",
578 nullptr);
579#endif
580 }

582 // When loading into a std::string or char*, accept a bytes/bytearray object as-is (i.e.
583 // without any encoding/decoding attempt). For other C++ char sizes this is a no-op.
584 // which supports loading a unicode from a str, doesn't take this path.
585 template <typename C = CharT>
586 bool load_raw(enable_if_t<std::is_same<C, char>::value, handle> src) {
587 if (PYBIND11_BYTES_CHECK(src.ptr())) {
588 // We were passed raw bytes; accept it into a std::string or char*
589 // without any encoding attempt.
590 const char *bytes = PYBIND11_BYTES_AS_STRING(src.ptr());
591 if (!bytes) {
592 pybind11_fail("Unexpected PYBIND11_BYTES_AS_STRING() failure.");
593 }
594 value = StringType(bytes, (size_t) PYBIND11_BYTES_SIZE(src.ptr()));
595 return true;
596 }
597 if (PyByteArray_Check(src.ptr())) {
598 // We were passed a bytearray; accept it into a std::string or char*
599 // without any encoding attempt.
600 const char *bytearray = PyByteArray_AsString(src.ptr());
601 if (!bytearray) {
602 pybind11_fail("Unexpected PyByteArray_AsString() failure.");
603 }
604 value = StringType(bytearray, (size_t) PyByteArray_Size(src.ptr()));
605 return true;
606 }
607
608 return false;
609 }

655 static handle cast(const CharT *src, return_value_policy policy, handle parent) {

662 static handle cast(CharT src, return_value_policy policy, handle parent) {
663 if (std::is_same<char, CharT>::value) {
664 handle s = PyUnicode_DecodeLatin1((const char *) &src, 1, nullptr);
665 if (!s) {
666 throw error_already_set();
667 }
668 return s;
669 }
670 return StringCaster::cast(StringType(1, src), policy, parent);
671 }

673 explicit operator CharT *() {
674 return none ? nullptr : const_cast<CharT *>(static_cast<StringType &>(str_caster).c_str());
675 }
676 explicit operator CharT &() {
677 if (none) {
678 throw value_error("Cannot convert None to a character");
679 }
680
681 auto &value = static_cast<StringType &>(str_caster);
682 size_t str_len = value.size();
683 if (str_len == 0) {
684 throw value_error("Cannot convert empty string to a character");
685 }
686
687 // If we're in UTF-8 mode, we have two possible failures: one for a unicode character that
688 // is too high, and one for multiple unicode characters (caught later), so we need to
689 // figure out how long the first encoded character is in bytes to distinguish between these
690 // two errors. We also allow want to allow unicode characters U+0080 through U+00FF, as
691 // those can fit into a single char value.
692 if (StringCaster::UTF_N == 8 && str_len > 1 && str_len <= 4) {
693 auto v0 = static_cast<unsigned char>(value[0]);
694 // low bits only: 0-127
695 // 0b110xxxxx - start of 2-byte sequence
696 // 0b1110xxxx - start of 3-byte sequence
697 // 0b11110xxx - start of 4-byte sequence
698 size_t char0_bytes = (v0 & 0x80) == 0 ? 1
699 : (v0 & 0xE0) == 0xC0 ? 2
700 : (v0 & 0xF0) == 0xE0 ? 3
701 : 4;
702
703 if (char0_bytes == str_len) {
704 // If we have a 128-255 value, we can decode it into a single char:
705 if (char0_bytes == 2 && (v0 & 0xFC) == 0xC0) { // 0x110000xx 0x10xxxxxx
706 one_char = static_cast<CharT>(((v0 & 3) << 6)
707 + (static_cast<unsigned char>(value[1]) & 0x3F));
708 return one_char;
709 }
710 // Otherwise we have a single character, but it's > U+00FF
711 throw value_error("Character code point not in range(0x100)");
712 }
713 }
714
715 // UTF-16 is much easier: we can only have a surrogate pair for values above U+FFFF, thus a
716 // surrogate pair with total length 2 instantly indicates a range error (but not a "your
717 // string was too long" error).
718 else if (StringCaster::UTF_N == 16 && str_len == 2) {
719 one_char = static_cast<CharT>(value[0]);
720 if (one_char >= 0xD800 && one_char < 0xE000) {
721 throw value_error("Character code point not in range(0x10000)");
722 }
723 }
724
725 if (str_len != 1) {
726 throw value_error("Expected a character, but multi-character string found");
727 }
728
729 one_char = value[0];
730 return one_char;
731 }

735 using cast_op_type = pybind11::detail::cast_op_type<_T>;

1531// Our conditions for enabling moving are quite restrictive:
1532// At compile time:
1533// - T needs to be a non-const, non-pointer, non-reference type
1534// - type_caster<T>::operator T&() must exist
1535// - the type must be move constructible (obviously)
1536// At run-time:
1537// - if the type is non-copy-constructible, the object must be the sole owner of the type (i.e. it
1538// must have ref_count() == 1)h
1539// If any of the above are not satisfied, we fall back to copying.
1540template <typename T>
1541using move_is_plain_type
1542 = satisfies_none_of<T, std::is_void, std::is_pointer, std::is_reference, std::is_const>;
1543template <typename T, typename SFINAE = void>
1544struct move_always : std::false_type {};
1545template <typename T>
1546struct move_always<
1547 T,
1548 enable_if_t<
1549 all_of<move_is_plain_type<T>,
1550 negation<is_copy_constructible<T>>,
1551 is_move_constructible<T>,
1552 std::is_same<decltype(std::declval<make_caster<T>>().operator T &()), T &>>::value>>
1553 : std::true_type {};
1554template <typename T, typename SFINAE = void>
1555struct move_if_unreferenced : std::false_type {};
1556template <typename T>
1557struct move_if_unreferenced<
1558 T,
1559 enable_if_t<
1560 all_of<move_is_plain_type<T>,
1561 negation<move_always<T>>,
1562 is_move_constructible<T>,
1563 std::is_same<decltype(std::declval<make_caster<T>>().operator T &()), T &>>::value>>
1564 : std::true_type {};
1565template <typename T>
1566using move_never = none_of<move_always<T>, move_if_unreferenced<T>>;

1568// Detect whether returning a `type` from a cast on type's type_caster is going to result in a
1569// reference or pointer to a local variable of the type_caster. Basically, only
1570// non-reference/pointer `type`s and reference/pointers from a type_caster_generic are safe;
1571// everything else returns a reference/pointer to a local variable.
1572template <typename type>
1573using cast_is_temporary_value_reference
1574 = bool_constant<(std::is_reference<type>::value || std::is_pointer<type>::value)
1575 && !std::is_base_of<type_caster_generic, make_caster<type>>::value
1576 && !std::is_same<intrinsic_t<type>, void>::value>;

1684template <typename T, detail::enable_if_t<!detail::is_pyobject<T>::value, int> = 0>
1685object cast(T &&value,
1686 return_value_policy policy = return_value_policy::automatic_reference,
1687 handle parent = handle()) {
1688 using no_ref_T = typename std::remove_reference<T>::type;
1689 if (policy == return_value_policy::automatic) {
1690 policy = std::is_pointer<no_ref_T>::value ? return_value_policy::take_ownership
1691 : std::is_lvalue_reference<T>::value ? return_value_policy::copy
1692 : return_value_policy::move;

1694 policy = std::is_pointer<no_ref_T>::value ? return_value_policy::reference
1695 : std::is_lvalue_reference<T>::value ? return_value_policy::copy
1696 : return_value_policy::move;
1697 }
1698 return reinterpret_steal<object>(
1699 detail::make_caster<T>::cast(std::forward<T>(value), policy, parent));
