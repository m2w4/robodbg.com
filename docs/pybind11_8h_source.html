63inline std::string replace_newlines_and_squash(const char *text) {
64 const char *whitespaces = " \t\n\r\f\v";
65 std::string result(text);
66 bool previous_is_whitespace = false;
67
68 if (result.size() >= 2) {
69 // Do not modify string representations
70 char first_char = result[0];
71 char last_char = result[result.size() - 1];
72 if (first_char == last_char && first_char == '\'') {
73 return result;
74 }
75 }
76 result.clear();
77
78 // Replace characters in whitespaces array with spaces and squash consecutive spaces
79 while (*text != '\0') {
80 if (std::strchr(whitespaces, *text)) {
81 if (!previous_is_whitespace) {
82 result += ' ';
83 previous_is_whitespace = true;
84 }
85 } else {
86 result += *text;
87 previous_is_whitespace = false;
88 }
89 ++text;
90 }
91
92 // Strip leading and trailing whitespaces
93 const size_t str_begin = result.find_first_not_of(whitespaces);
94 if (str_begin == std::string::npos) {
95 return "";
96 }
97
98 const size_t str_end = result.find_last_not_of(whitespaces);
99 const size_t str_range = str_end - str_begin + 1;
100
101 return result.substr(str_begin, str_range);
102}

104/* Generate a proper function signature */
105inline std::string generate_function_signature(const char *type_caster_name_field,
106 detail::function_record *func_rec,
107 const std::type_info *const *types,
108 size_t &type_index,
109 size_t &arg_index) {
110 std::string signature;
111 bool is_starred = false;
112 // `is_return_value.top()` is true if we are currently inside the return type of the
113 // signature. Using `@^`/`@$` we can force types to be arg/return types while `@!` pops
114 // back to the previous state.
115 std::stack<bool> is_return_value({false});
116 // The following characters have special meaning in the signature parsing. Literals
117 // containing these are escaped with `!`.
118 std::string special_chars("!@%{}-");
119 for (const auto *pc = type_caster_name_field; *pc != '\0'; ++pc) {
120 const auto c = *pc;
121 if (c == '{') {
122 // Write arg name for everything except *args and **kwargs.
123 is_starred = *(pc + 1) == '*';
124 if (is_starred) {
125 continue;
126 }
127 // Separator for keyword-only arguments, placed before the kw
128 // arguments start (unless we are already putting an *args)
129 if (!func_rec->has_args && arg_index == func_rec->nargs_pos) {
130 signature += "*, ";
131 }
132 if (arg_index < func_rec->args.size() && func_rec->args[arg_index].name) {
133 signature += func_rec->args[arg_index].name;
134 } else if (arg_index == 0 && func_rec->is_method) {
135 signature += "self";
136 } else {
137 signature += "arg" + std::to_string(arg_index - (func_rec->is_method ? 1 : 0));
138 }
139 signature += ": ";
140 } else if (c == '}') {
141 // Write default value if available.
142 if (!is_starred && arg_index < func_rec->args.size()
143 && func_rec->args[arg_index].descr) {
144 signature += " = ";
145 signature += detail::replace_newlines_and_squash(func_rec->args[arg_index].descr);
146 }
147 // Separator for positional-only arguments (placed after the
148 // argument, rather than before like *
149 if (func_rec->nargs_pos_only > 0 && (arg_index + 1) == func_rec->nargs_pos_only) {
150 signature += ", /";
151 }
152 if (!is_starred) {
153 arg_index++;
154 }
155 } else if (c == '%') {
156 const std::type_info *t = types[type_index++];
157 if (!t) {
158 pybind11_fail("Internal error while parsing type signature (1)");
159 }
160 if (auto *tinfo = detail::get_type_info(*t)) {
161 handle th((PyObject *) tinfo->type);
162 signature += th.attr("__module__").cast<std::string>() + "."
163 + th.attr("__qualname__").cast<std::string>();
164 } else if (auto th = detail::global_internals_native_enum_type_map_get_item(*t)) {
165 signature += th.attr("__module__").cast<std::string>() + "."
166 + th.attr("__qualname__").cast<std::string>();
167 } else if (func_rec->is_new_style_constructor && arg_index == 0) {
168 // A new-style `__init__` takes `self` as `value_and_holder`.
169 // Rewrite it to the proper class type.
170 signature += func_rec->scope.attr("__module__").cast<std::string>() + "."
171 + func_rec->scope.attr("__qualname__").cast<std::string>();
172 } else {
173 signature += detail::quote_cpp_type_name(detail::clean_type_id(t->name()));
174 }
175 } else if (c == '!' && special_chars.find(*(pc + 1)) != std::string::npos) {
176 // typing::Literal escapes special characters with !
177 signature += *++pc;
178 } else if (c == '@') {
179 // `@^ ... @!` and `@$ ... @!` are used to force arg/return value type (see
180 // typing::Callable/detail::arg_descr/detail::return_descr)
181 if (*(pc + 1) == '^') {
182 is_return_value.emplace(false);
183 ++pc;
184 continue;
185 }
186 if (*(pc + 1) == '$') {
187 is_return_value.emplace(true);
188 ++pc;
189 continue;
190 }
191 if (*(pc + 1) == '!') {
192 is_return_value.pop();
193 ++pc;
194 continue;
195 }
196 // Handle types that differ depending on whether they appear
197 // in an argument or a return value position (see io_name<text1, text2>).
198 // For named arguments (py::arg()) with noconvert set, return value type is used.
199 ++pc;
200 if (!is_return_value.top()
201 && (!(arg_index < func_rec->args.size() && !func_rec->args[arg_index].convert))) {
202 while (*pc != '\0' && *pc != '@') {
203 signature += *pc++;
204 }
205 if (*pc == '@') {
206 ++pc;
207 }
208 while (*pc != '\0' && *pc != '@') {
209 ++pc;
210 }
211 } else {
212 while (*pc != '\0' && *pc != '@') {
213 ++pc;
214 }
215 if (*pc == '@') {
216 ++pc;
217 }
218 while (*pc != '\0' && *pc != '@') {
219 signature += *pc++;
220 }
221 }
222 } else {
223 if (c == '-' && *(pc + 1) == '>') {
224 is_return_value.emplace(true);
225 }
226 signature += c;
227 }
228 }
229 return signature;
230}

232template <typename T>
233inline std::string generate_type_signature() {
234 static constexpr auto caster_name_field = make_caster<T>::name;
235 PYBIND11_DESCR_CONSTEXPR auto descr_types = decltype(caster_name_field)::types();
236 // Create a default function_record to ensure the function signature has the proper
237 // configuration e.g. no_convert.
238 auto func_rec = function_record();
239 size_t type_index = 0;
240 size_t arg_index = 0;
241 return generate_function_signature(
242 caster_name_field.text, &func_rec, descr_types.data(), type_index, arg_index);
243}

340 template <typename Func, typename Return, typename... Args, typename... Extra>
341 void initialize(Func &&f, Return (*)(Args...), const Extra &...extra) {
342 using namespace detail;
343 struct capture {
344 remove_reference_t<Func> f;
345
346 static capture *from_data(void **data) {
347 return PYBIND11_STD_LAUNDER(reinterpret_cast<capture *>(data));
348 }
349 };
350
351 /* Store the function including any extra state it might have (e.g. a lambda capture
352 * object) */
353 // The unique_ptr makes sure nothing is leaked in case of an exception.
354 auto unique_rec = make_function_record();
355 auto *rec = unique_rec.get();
356
357 /* Store the capture object directly in the function record if there is enough space */
358 if (sizeof(capture) <= sizeof(rec->data)) {
359 /* Without these pragmas, GCC warns that there might not be
360 enough space to use the placement new operator. However, the
361 'if' statement above ensures that this is the case. */
362 PYBIND11_WARNING_PUSH
363
364#if defined(__GNUG__) && __GNUC__ >= 6
365 PYBIND11_WARNING_DISABLE_GCC("-Wplacement-new")
366#endif
367
368 new (capture::from_data(rec->data)) capture{std::forward<Func>(f)};
369
370#if !PYBIND11_HAS_STD_LAUNDER
371 PYBIND11_WARNING_DISABLE_GCC("-Wstrict-aliasing")
372#endif
373
374 // UB without std::launder, but without breaking ABI and/or
375 // a significant refactoring it's "impossible" to solve.
376 if (!std::is_trivially_destructible<capture>::value) {
377 rec->free_data = [](function_record *r) {
378 auto data = capture::from_data(r->data);
379 (void) data; // suppress "unused variable" warnings
380 data->~capture();
381 };
382 }
383 PYBIND11_WARNING_POP
384 } else {
385 rec->data[0] = new capture{std::forward<Func>(f)};
386 rec->free_data = [](function_record *r) { delete ((capture *) r->data[0]); };
387 }
388
389 /* Type casters for the function arguments and return value */
390 using cast_in = argument_loader<Args...>;
391 using cast_out
392 = make_caster<conditional_t<std::is_void<Return>::value, void_type, Return>>;
393
394 static_assert(
395 expected_num_args<Extra...>(
396 sizeof...(Args), cast_in::args_pos >= 0, cast_in::has_kwargs),
397 "The number of argument annotations does not match the number of function arguments");
398
399 /* Dispatch code which converts function arguments and performs the actual function call */
400 rec->impl = [](function_call &call) -> handle {
401 cast_in args_converter;
402
403 /* Try to cast the function arguments into the C++ domain */
404 if (!args_converter.load_args(call)) {
405 return PYBIND11_TRY_NEXT_OVERLOAD;
406 }
407
408 /* Invoke call policy pre-call hook */
409 process_attributes<Extra...>::precall(call);
410
411 /* Get a pointer to the capture object */
412 const auto *data = (sizeof(capture) <= sizeof(call.func.data) ? &call.func.data
413 : call.func.data[0]);
414 auto *cap = const_cast<capture *>(reinterpret_cast<const capture *>(data));
415
416 /* Override policy for rvalues -- usually to enforce rvp::move on an rvalue */
417 return_value_policy policy
418 = return_value_policy_override<Return>::policy(call.func.policy);
419
420 /* Function scope guard -- defaults to the compile-to-nothing `void_type` */
421 using Guard = extract_guard_t<Extra...>;
422
423 /* Perform the function call */
424 handle result;
425 if (call.func.is_setter) {
426 (void) std::move(args_converter).template call<Return, Guard>(cap->f);
427 result = none().release();
428 } else {
429 result = cast_out::cast(
430 std::move(args_converter).template call<Return, Guard>(cap->f),
431 policy,
432 call.parent);
433 }
434
435 /* Invoke call policy post-call hook */
436 process_attributes<Extra...>::postcall(call, result);
437
438 return result;
439 };
440
441 rec->nargs_pos = cast_in::args_pos >= 0
442 ? static_cast<std::uint16_t>(cast_in::args_pos)
443 : sizeof...(Args) - cast_in::has_kwargs; // Will get reduced more if
444 // we have a kw_only
445 rec->has_args = cast_in::args_pos >= 0;
446 rec->has_kwargs = cast_in::has_kwargs;
447
448 /* Process any user-provided function attributes */
449 process_attributes<Extra...>::init(extra..., rec);
450
451 {
452 constexpr bool has_kw_only_args = any_of<std::is_same<kw_only, Extra>...>::value,
453 has_pos_only_args = any_of<std::is_same<pos_only, Extra>...>::value,
454 has_arg_annotations = any_of<is_keyword<Extra>...>::value;
455 constexpr bool has_is_method = any_of<std::is_same<is_method, Extra>...>::value;
456 // The implicit `self` argument is not present and not counted in method definitions.
457 constexpr bool has_args = cast_in::args_pos >= 0;
458 constexpr bool is_method_with_self_arg_only = has_is_method && !has_args;
459 static_assert(has_arg_annotations || !has_kw_only_args,
460 "py::kw_only requires the use of argument annotations");
461 static_assert(((/* Need `py::arg("arg_name")` annotation in function/method. */
462 has_arg_annotations)
463 || (/* Allow methods with no arguments `def method(self, /): ...`.
464 * A method has at least one argument `self`. There can be no
465 * `py::arg` annotation. E.g. `class.def("method", py::pos_only())`.
466 */
467 is_method_with_self_arg_only))
468 || !has_pos_only_args,
469 "py::pos_only requires the use of argument annotations (for docstrings "
470 "and aligning the annotations to the argument)");
471
472 static_assert(constexpr_sum(is_kw_only<Extra>::value...) <= 1,
473 "py::kw_only may be specified only once");
474 static_assert(constexpr_sum(is_pos_only<Extra>::value...) <= 1,
475 "py::pos_only may be specified only once");
476 constexpr auto kw_only_pos = constexpr_first<is_kw_only, Extra...>();
477 constexpr auto pos_only_pos = constexpr_first<is_pos_only, Extra...>();
478 static_assert(!(has_kw_only_args && has_pos_only_args) || pos_only_pos < kw_only_pos,
479 "py::pos_only must come before py::kw_only");
480 }
481
482 /* Generate a readable signature describing the function's arguments and return
483 value types */
484 static constexpr auto signature
485 = const_name("(") + cast_in::arg_names + const_name(") -> ") + cast_out::name;
486 PYBIND11_DESCR_CONSTEXPR auto types = decltype(signature)::types();
487
488 /* Register the function with Python from generic (non-templated) code */
489 // Pass on the ownership over the `unique_rec` to `initialize_generic`. `rec` stays valid.
490 initialize_generic(std::move(unique_rec), signature.text, types.data(), sizeof...(Args));
491
492 /* Stash some additional information used by an important optimization in 'functional.h' */
493 using FunctionType = Return (*)(Args...);
494 constexpr bool is_function_ptr
495 = std::is_convertible<Func, FunctionType>::value && sizeof(capture) == sizeof(void *);
496 PYBIND11_ENSURE_PRECONDITION_FOR_FUNCTIONAL_H_PERFORMANCE_OPTIMIZATIONS(
497 !is_function_ptr || std::is_standard_layout<capture>::value);
498 if (is_function_ptr) {
499 rec->is_stateless = true;
500 rec->data[1]
501 = const_cast<void *>(reinterpret_cast<const void *>(&typeid(FunctionType)));
502 }
503 }

505 // Utility class that keeps track of all duplicated strings, and cleans them up in its
506 // destructor, unless they are released. Basically a RAII-solution to deal with exceptions
507 // along the way.
508 class strdup_guard {
509 public:
510 strdup_guard() = default;
511 strdup_guard(const strdup_guard &) = delete;
512 strdup_guard &operator=(const strdup_guard &) = delete;
513
514 ~strdup_guard() {
515 for (auto *s : strings) {
516 std::free(s);
517 }
518 }
519 char *operator()(const char *s) {
520 auto *t = PYBIND11_COMPAT_STRDUP(s);
521 strings.push_back(t);
522 return t;
523 }
524 void release() { strings.clear(); }
525
526 private:
527 std::vector<char *> strings;
528 };

530 /// Register a function call with Python (generic non-templated code goes here)
531 void initialize_generic(unique_function_record &&unique_rec,
532 const char *text,
533 const std::type_info *const *types,
534 size_t args) {
535 // Do NOT receive `unique_rec` by value. If this function fails to move out the unique_ptr,
536 // we do not want this to destruct the pointer. `initialize` (the caller) still relies on
537 // the pointee being alive after this call. Only move out if a `capsule` is going to keep
538 // it alive.
539 auto *rec = unique_rec.get();
540
541 // Keep track of strdup'ed strings, and clean them up as long as the function's capsule
542 // has not taken ownership yet (when `unique_rec.release()` is called).
543 // Note: This cannot easily be fixed by a `unique_ptr` with custom deleter, because the
544 // strings are only referenced before strdup'ing. So only *after* the following block could
545 // `destruct` safely be called, but even then, `repr` could still throw in the middle of
546 // copying all strings.
547 strdup_guard guarded_strdup;
548
549 /* Create copies of all referenced C-style strings */
550 rec->name = guarded_strdup(rec->name ? rec->name : "");
551 if (rec->doc) {
552 rec->doc = guarded_strdup(rec->doc);
553 }
554 for (auto &a : rec->args) {
555 if (a.name) {
556 a.name = guarded_strdup(a.name);
557 }
558 if (a.descr) {
559 a.descr = guarded_strdup(a.descr);
560 } else if (a.value) {
561 a.descr = guarded_strdup(repr(a.value).cast<std::string>().c_str());
562 }
563 }
564
565 rec->is_constructor = (std::strcmp(rec->name, "__init__") == 0)
566 || (std::strcmp(rec->name, "__setstate__") == 0);
567
568#if defined(PYBIND11_DETAILED_ERROR_MESSAGES) && !defined(PYBIND11_DISABLE_NEW_STYLE_INIT_WARNING)
569 if (rec->is_constructor && !rec->is_new_style_constructor) {
570 const auto class_name
571 = detail::get_fully_qualified_tp_name((PyTypeObject *) rec->scope.ptr());
572 const auto func_name = std::string(rec->name);
573 PyErr_WarnEx(PyExc_FutureWarning,
574 ("pybind11-bound class '" + class_name
575 + "' is using an old-style "
576 "placement-new '"
577 + func_name
578 + "' which has been deprecated. See "
579 "the upgrade guide in pybind11's docs. This message is only visible "
580 "when compiled in debug mode.")
581 .c_str(),
582 0);
583 }
584#endif
585
586 size_t type_index = 0, arg_index = 0;
587 std::string signature
588 = detail::generate_function_signature(text, rec, types, type_index, arg_index);
589
590 if (arg_index != args - rec->has_args - rec->has_kwargs || types[type_index] != nullptr) {
591 pybind11_fail("Internal error while parsing type signature (2)");
592 }
593
594 rec->signature = guarded_strdup(signature.c_str());
595 rec->args.shrink_to_fit();
596 rec->nargs = (std::uint16_t) args;
597
598 if (rec->sibling && PYBIND11_INSTANCE_METHOD_CHECK(rec->sibling.ptr())) {
599 rec->sibling = PYBIND11_INSTANCE_METHOD_GET_FUNCTION(rec->sibling.ptr());
600 }
601
602 detail::function_record *chain = nullptr, *chain_start = rec;
603 if (rec->sibling) {
604 if (PyCFunction_Check(rec->sibling.ptr())) {
605 auto *self = PyCFunction_GET_SELF(rec->sibling.ptr());
606 if (self == nullptr) {
607 pybind11_fail(
608 "initialize_generic: Unexpected nullptr from PyCFunction_GET_SELF");
609 }
610 chain = detail::function_record_ptr_from_PyObject(self);
611 if (chain && !chain->scope.is(rec->scope)) {
612 /* Never append a method to an overload chain of a parent class;
613 instead, hide the parent's overloads in this case */
614 chain = nullptr;
615 }
616 }
617 // Don't trigger for things like the default __init__, which are wrapper_descriptors
618 // that we are intentionally replacing
619 else if (!rec->sibling.is_none() && rec->name[0] != '_') {
620 pybind11_fail("Cannot overload existing non-function object \""
621 + std::string(rec->name) + "\" with a function of the same name");
622 }
623 }
624
625 if (!chain) {
626 /* No existing overload was found, create a new function object */
627 rec->def = new PyMethodDef();
628 std::memset(rec->def, 0, sizeof(PyMethodDef));
629 rec->def->ml_name = rec->name;
630 rec->def->ml_meth
631 = reinterpret_cast<PyCFunction>(reinterpret_cast<void (*)()>(dispatcher));
632 rec->def->ml_flags = METH_VARARGS | METH_KEYWORDS;
633
634 object py_func_rec = detail::function_record_PyObject_New();
635 ((detail::function_record_PyObject *) py_func_rec.ptr())->cpp_func_rec
636 = unique_rec.release();
637 guarded_strdup.release();
638
639 object scope_module = detail::get_scope_module(rec->scope);
640 m_ptr = PyCFunction_NewEx(rec->def, py_func_rec.ptr(), scope_module.ptr());
641 if (!m_ptr) {
642 pybind11_fail("cpp_function::cpp_function(): Could not allocate function object");
643 }
644 } else {
645 /* Append at the beginning or end of the overload chain */
646 m_ptr = rec->sibling.ptr();
647 inc_ref();
648 if (chain->is_method != rec->is_method) {
649 pybind11_fail(
650 "overloading a method with both static and instance methods is not supported; "
651#if !defined(PYBIND11_DETAILED_ERROR_MESSAGES)
652 "#define PYBIND11_DETAILED_ERROR_MESSAGES or compile in debug mode for more "
653 "details"
654#else
655 "error while attempting to bind "
656 + std::string(rec->is_method ? "instance" : "static") + " method "
657 + std::string(pybind11::str(rec->scope.attr("__name__"))) + "."
658 + std::string(rec->name) + signature
659#endif
660 );
661 }
662
663 if (rec->prepend) {
664 // Beginning of chain; we need to replace the capsule's current head-of-the-chain
665 // pointer with this one, then make this one point to the previous head of the
666 // chain.
667 chain_start = rec;
668 rec->next = chain;
669 auto *py_func_rec
670 = (detail::function_record_PyObject *) PyCFunction_GET_SELF(m_ptr);
671 py_func_rec->cpp_func_rec = unique_rec.release();
672 guarded_strdup.release();
673 } else {
674 // Or end of chain (normal behavior)
675 chain_start = chain;
676 while (chain->next) {
677 chain = chain->next;
678 }
679 chain->next = unique_rec.release();
680 guarded_strdup.release();
681 }
682 }
683
684 std::string signatures;
685 int index = 0;
686 /* Create a nice pydoc rec including all signatures and
687 docstrings of the functions in the overload chain */
688 if (chain && options::show_function_signatures()
689 && std::strcmp(rec->name, "_pybind11_conduit_v1_") != 0) {
690 // First a generic signature
691 signatures += rec->name;
692 signatures += "(*args, **kwargs)\n";
693 signatures += "Overloaded function.\n\n";
694 }
695 // Then specific overload signatures
696 bool first_user_def = true;
697 for (auto *it = chain_start; it != nullptr; it = it->next) {
698 if (options::show_function_signatures()
699 && std::strcmp(rec->name, "_pybind11_conduit_v1_") != 0) {
700 if (index > 0) {
701 signatures += '\n';
702 }
703 if (chain) {
704 signatures += std::to_string(++index) + ". ";
705 }
706 signatures += rec->name;
707 signatures += it->signature;
708 signatures += '\n';
709 }
710 if (it->doc && it->doc[0] != '\0' && options::show_user_defined_docstrings()) {
711 // If we're appending another docstring, and aren't printing function signatures,
712 // we need to append a newline first:
713 if (!options::show_function_signatures()) {
714 if (first_user_def) {
715 first_user_def = false;
716 } else {
717 signatures += '\n';
718 }
719 }
720 if (options::show_function_signatures()) {
721 signatures += '\n';
722 }
723 signatures += it->doc;
724 if (options::show_function_signatures()) {
725 signatures += '\n';
726 }
727 }
728 }
729
730 auto *func = (PyCFunctionObject *) m_ptr;
731 // Install docstring if it's non-empty (when at least one option is enabled)
732 auto *doc = signatures.empty() ? nullptr : PYBIND11_COMPAT_STRDUP(signatures.c_str());
733 std::free(const_cast<char *>(PYBIND11_PYCFUNCTION_GET_DOC(func)));
734 PYBIND11_PYCFUNCTION_SET_DOC(func, doc);
735
736 if (rec->is_method) {
737 m_ptr = PYBIND11_INSTANCE_METHOD_NEW(m_ptr, rec->scope.ptr());
738 if (!m_ptr) {
739 pybind11_fail(
740 "cpp_function::cpp_function(): Could not allocate instance method object");
741 }
742 Py_DECREF(func);
743 }
744 }

748 /// When a cpp_function is GCed, release any memory allocated by pybind11
749 static void destruct(detail::function_record *rec, bool free_strings = true) {
750// If on Python 3.9, check the interpreter "MICRO" (patch) version.
751// If this is running on 3.9.0, we have to work around a bug.
752#if !defined(PYPY_VERSION) && PY_MAJOR_VERSION == 3 && PY_MINOR_VERSION == 9
753 static bool is_zero = Py_GetVersion()[4] == '0';
754#endif
755
756 while (rec) {
757 detail::function_record *next = rec->next;
758 if (rec->free_data) {
759 rec->free_data(rec);
760 }
761 // During initialization, these strings might not have been copied yet,
762 // so they cannot be freed. Once the function has been created, they can.
763 // Check `make_function_record` for more details.
764 if (free_strings) {
765 std::free((char *) rec->name);
766 std::free((char *) rec->doc);
767 std::free((char *) rec->signature);
768 for (auto &arg : rec->args) {
769 std::free(const_cast<char *>(arg.name));
770 std::free(const_cast<char *>(arg.descr));
771 }
772 }
773 for (auto &arg : rec->args) {
774 arg.value.dec_ref();
775 }
776 if (rec->def) {
777 std::free(const_cast<char *>(rec->def->ml_doc));
778// Python 3.9.0 decref's these in the wrong order; rec->def
779// If loaded on 3.9.0, let these leak (use Python 3.9.1 at runtime to fix)
780// See https://github.com/python/cpython/pull/22670
781#if !defined(PYPY_VERSION) && PY_MAJOR_VERSION == 3 && PY_MINOR_VERSION == 9
782 if (!is_zero) {
783 delete rec->def;
784 }
785#else
786 delete rec->def;
787#endif
788 }
789 delete rec;
790 rec = next;
791 }
792 }

794 /// Main dispatch logic for calls to functions bound using pybind11
795 static PyObject *dispatcher(PyObject *self, PyObject *args_in, PyObject *kwargs_in) {
796 using namespace detail;
797 const function_record *overloads = function_record_ptr_from_PyObject(self);
798 assert(overloads != nullptr);
799
800 /* Iterator over the list of potentially admissible overloads */
801 const function_record *current_overload = overloads;
802
803 /* Need to know how many arguments + keyword arguments there are to pick the right
804 overload */
805 const auto n_args_in = (size_t) PyTuple_GET_SIZE(args_in);
806
807 handle parent = n_args_in > 0 ? PyTuple_GET_ITEM(args_in, 0) : nullptr,
808 result = PYBIND11_TRY_NEXT_OVERLOAD;
809
810 auto self_value_and_holder = value_and_holder();
811 if (overloads->is_constructor) {
812 if (!parent
813 || !PyObject_TypeCheck(parent.ptr(), (PyTypeObject *) overloads->scope.ptr())) {
814 set_error(PyExc_TypeError,
815 "__init__(self, ...) called with invalid or missing `self` argument");
816 return nullptr;
817 }
818
819 auto *const tinfo = get_type_info((PyTypeObject *) overloads->scope.ptr());
820 auto *const pi = reinterpret_cast<instance *>(parent.ptr());
821 self_value_and_holder = pi->get_value_and_holder(tinfo, true);
822
823 // If this value is already registered it must mean __init__ is invoked multiple times;
824 // we really can't support that in C++, so just ignore the second __init__.
825 if (self_value_and_holder.instance_registered()) {
826 return none().release().ptr();
827 }
828 }
829
830 try {
831 // We do this in two passes: in the first pass, we load arguments with `convert=false`;
832 // in the second, we allow conversion (except for arguments with an explicit
833 // py::arg().noconvert()). This lets us prefer calls without conversion, with
834 // conversion as a fallback.
835 std::vector<function_call> second_pass;
836
837 // However, if there are no overloads, we can just skip the no-convert pass entirely
838 const bool overloaded
839 = current_overload != nullptr && current_overload->next != nullptr;
840
841 for (; current_overload != nullptr; current_overload = current_overload->next) {
842
843 /* For each overload:
844 1. Copy all positional arguments we were given, also checking to make sure that
845 named positional arguments weren't *also* specified via kwarg.
846 2. If we weren't given enough, try to make up the omitted ones by checking
847 whether they were provided by a kwarg matching the `py::arg("name")` name. If
848 so, use it (and remove it from kwargs); if not, see if the function binding
849 provided a default that we can use.
850 3. Ensure that either all keyword arguments were "consumed", or that the
851 function takes a kwargs argument to accept unconsumed kwargs.
852 4. Any positional arguments still left get put into a tuple (for args), and any
853 leftover kwargs get put into a dict.
854 5. Pack everything into a vector; if we have py::args or py::kwargs, they are an
855 extra tuple or dict at the end of the positional arguments.
856 6. Call the function call dispatcher (function_record::impl)
857
858 If one of these fail, move on to the next overload and keep trying until we get
859 a result other than PYBIND11_TRY_NEXT_OVERLOAD.
860 */
861
862 const function_record &func = *current_overload;
863 size_t num_args = func.nargs; // Number of positional arguments that we need
864 if (func.has_args) {
865 --num_args; // (but don't count py::args
866 }
867 if (func.has_kwargs) {
868 --num_args; // or py::kwargs)
869 }
870 size_t pos_args = func.nargs_pos;
871
872 if (!func.has_args && n_args_in > pos_args) {
873 continue; // Too many positional arguments for this overload
874 }
875
876 if (n_args_in < pos_args && func.args.size() < pos_args) {
877 continue; // Not enough positional arguments given, and not enough defaults to
878 // fill in the blanks
879 }
880
881 function_call call(func, parent);
882
883 // Protect std::min with parentheses
884 size_t args_to_copy = (std::min) (pos_args, n_args_in);
885 size_t args_copied = 0;
886
887 // 0. Inject new-style `self` argument
888 if (func.is_new_style_constructor) {
889 // The `value` may have been preallocated by an old-style `__init__`
890 // if it was a preceding candidate for overload resolution.
891 if (self_value_and_holder) {
892 self_value_and_holder.type->dealloc(self_value_and_holder);
893 }
894
895 call.init_self = PyTuple_GET_ITEM(args_in, 0);
896 call.args.emplace_back(reinterpret_cast<PyObject *>(&self_value_and_holder));
897 call.args_convert.push_back(false);
898 ++args_copied;
899 }
900
901 // 1. Copy any position arguments given.
902 bool bad_arg = false;
903 for (; args_copied < args_to_copy; ++args_copied) {
904 const argument_record *arg_rec
905 = args_copied < func.args.size() ? &func.args[args_copied] : nullptr;
906 if (kwargs_in && arg_rec && arg_rec->name
907 && dict_getitemstring(kwargs_in, arg_rec->name)) {
908 bad_arg = true;
909 break;
910 }
911
912 handle arg(PyTuple_GET_ITEM(args_in, args_copied));
913 if (arg_rec && !arg_rec->none && arg.is_none()) {
914 bad_arg = true;
915 break;
916 }
917 call.args.push_back(arg);
918 call.args_convert.push_back(arg_rec ? arg_rec->convert : true);
919 }
920 if (bad_arg) {
921 continue; // Maybe it was meant for another overload (issue #688)
922 }
923
924 // Keep track of how many position args we copied out in case we need to come back
925 // to copy the rest into a py::args argument.
926 size_t positional_args_copied = args_copied;
927
928 // We'll need to copy this if we steal some kwargs for defaults
929 dict kwargs = reinterpret_borrow<dict>(kwargs_in);
930
931 // 1.5. Fill in any missing pos_only args from defaults if they exist
932 if (args_copied < func.nargs_pos_only) {
933 for (; args_copied < func.nargs_pos_only; ++args_copied) {
934 const auto &arg_rec = func.args[args_copied];
935 handle value;
936
937 if (arg_rec.value) {
938 value = arg_rec.value;
939 }
940 if (value) {
941 call.args.push_back(value);
942 call.args_convert.push_back(arg_rec.convert);
943 } else {
944 break;
945 }
946 }
947
948 if (args_copied < func.nargs_pos_only) {
949 continue; // Not enough defaults to fill the positional arguments
950 }
951 }
952
953 // 2. Check kwargs and, failing that, defaults that may help complete the list
954 if (args_copied < num_args) {
955 bool copied_kwargs = false;
956
957 for (; args_copied < num_args; ++args_copied) {
958 const auto &arg_rec = func.args[args_copied];
959
960 handle value;
961 if (kwargs_in && arg_rec.name) {
962 value = dict_getitemstring(kwargs.ptr(), arg_rec.name);
963 }
964
965 if (value) {
966 // Consume a kwargs value
967 if (!copied_kwargs) {
968 kwargs = reinterpret_steal<dict>(PyDict_Copy(kwargs.ptr()));
969 copied_kwargs = true;
970 }
971 if (PyDict_DelItemString(kwargs.ptr(), arg_rec.name) == -1) {
972 throw error_already_set();
973 }
974 } else if (arg_rec.value) {
975 value = arg_rec.value;
976 }
977
978 if (!arg_rec.none && value.is_none()) {
979 break;
980 }
981
982 if (value) {
983 // If we're at the py::args index then first insert a stub for it to be
984 // replaced later
985 if (func.has_args && call.args.size() == func.nargs_pos) {
986 call.args.push_back(none());
987 }
988
989 call.args.push_back(value);
990 call.args_convert.push_back(arg_rec.convert);
991 } else {
992 break;
993 }
994 }
995
996 if (args_copied < num_args) {
997 continue; // Not enough arguments, defaults, or kwargs to fill the
998 // positional arguments
999 }
1000 }
1001
1002 // 3. Check everything was consumed (unless we have a kwargs arg)
1003 if (kwargs && !kwargs.empty() && !func.has_kwargs) {
1004 continue; // Unconsumed kwargs, but no py::kwargs argument to accept them
1005 }
1006
1007 // 4a. If we have a py::args argument, create a new tuple with leftovers
1008 if (func.has_args) {
1009 tuple extra_args;
1010 if (args_to_copy == 0) {
1011 // We didn't copy out any position arguments from the args_in tuple, so we
1012 // can reuse it directly without copying:
1013 extra_args = reinterpret_borrow<tuple>(args_in);
1014 } else if (positional_args_copied >= n_args_in) {
1015 extra_args = tuple(0);
1016 } else {
1017 size_t args_size = n_args_in - positional_args_copied;
1018 extra_args = tuple(args_size);
1019 for (size_t i = 0; i < args_size; ++i) {
1020 extra_args[i] = PyTuple_GET_ITEM(args_in, positional_args_copied + i);
1021 }
1022 }
1023 if (call.args.size() <= func.nargs_pos) {
1024 call.args.push_back(extra_args);
1025 } else {
1026 call.args[func.nargs_pos] = extra_args;
1027 }
1028 call.args_convert.push_back(false);
1029 call.args_ref = std::move(extra_args);
1030 }
1031
1032 // 4b. If we have a py::kwargs, pass on any remaining kwargs
1033 if (func.has_kwargs) {
1034 if (!kwargs.ptr()) {
1035 kwargs = dict(); // If we didn't get one, send an empty one
1036 }
1037 call.args.push_back(kwargs);
1038 call.args_convert.push_back(false);
1039 call.kwargs_ref = std::move(kwargs);
1040 }
1041
1042// 5. Put everything in a vector. Not technically step 5, we've been building it
1043// in `call.args` all along.
1044#if defined(PYBIND11_DETAILED_ERROR_MESSAGES)
1045 if (call.args.size() != func.nargs || call.args_convert.size() != func.nargs) {
1046 pybind11_fail("Internal error: function call dispatcher inserted wrong number "
1047 "of arguments!");
1048 }
1049#endif
1050
1051 std::vector<bool> second_pass_convert;
1052 if (overloaded) {
1053 // We're in the first no-convert pass, so swap out the conversion flags for a
1054 // set of all-false flags. If the call fails, we'll swap the flags back in for
1055 // the conversion-allowed call below.
1056 second_pass_convert.resize(func.nargs, false);
1057 call.args_convert.swap(second_pass_convert);
1058 }
1059
1060 // 6. Call the function.
1061 try {
1062 loader_life_support guard{};
1063 result = func.impl(call);
1064 } catch (reference_cast_error &) {
1065 result = PYBIND11_TRY_NEXT_OVERLOAD;
1066 }
1067
1068 if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD) {
1069 break;
1070 }
1071
1072 if (overloaded) {
1073 // The (overloaded) call failed; if the call has at least one argument that
1074 // permits conversion (i.e. it hasn't been explicitly specified `.noconvert()`)
1075 // then add this call to the list of second pass overloads to try.
1076 for (size_t i = func.is_method ? 1 : 0; i < pos_args; i++) {
1077 if (second_pass_convert[i]) {
1078 // Found one: swap the converting flags back in and store the call for
1079 // the second pass.
1080 call.args_convert.swap(second_pass_convert);
1081 second_pass.push_back(std::move(call));
1082 break;
1083 }
1084 }
1085 }
1086 }
1087
1088 if (overloaded && !second_pass.empty() && result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
1089 // The no-conversion pass finished without success, try again with conversion
1090 // allowed
1091 for (auto &call : second_pass) {
1092 try {
1093 loader_life_support guard{};
1094 result = call.func.impl(call);
1095 } catch (reference_cast_error &) {
1096 result = PYBIND11_TRY_NEXT_OVERLOAD;
1097 }
1098
1099 if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD) {
1100 // The error reporting logic below expects 'current_overload' to be valid,
1101 // as it would be if we'd encountered this failure in the first-pass loop.
1102 if (!result) {
1103 current_overload = &call.func;
1104 }
1105 break;
1106 }
1107 }
1108 }
1109 } catch (error_already_set &e) {
1110 e.restore();
1111 return nullptr;
1112#ifdef __GLIBCXX__
1113 } catch (abi::__forced_unwind &) {
1114 throw;
1115#endif
1116 } catch (...) {
1117 try_translate_exceptions();
1118 return nullptr;
1119 }
1120
1121 auto append_note_if_missing_header_is_suspected = [](std::string &msg) {
1122 if (msg.find("std::") != std::string::npos) {
1123 msg += "\n\n"
1124 "Did you forget to `#include <pybind11/stl.h>`? Or <pybind11/complex.h>,\n"
1125 "<pybind11/functional.h>, <pybind11/chrono.h>, etc. Some automatic\n"
1126 "conversions are optional and require extra headers to be included\n"
1127 "when compiling your pybind11 module.";
1128 }
1129 };
1130
1131 if (result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
1132 if (overloads->is_operator) {
1133 return handle(Py_NotImplemented).inc_ref().ptr();
1134 }
1135
1136 std::string msg = std::string(overloads->name) + "(): incompatible "
1137 + std::string(overloads->is_constructor ? "constructor" : "function")
1138 + " arguments. The following argument types are supported:\n";
1139
1140 int ctr = 0;
1141 for (const function_record *it2 = overloads; it2 != nullptr; it2 = it2->next) {
1142 msg += " " + std::to_string(++ctr) + ". ";
1143
1144 bool wrote_sig = false;
1145 if (overloads->is_constructor) {
1146 // For a constructor, rewrite `(self: Object, arg0, ...) -> NoneType` as
1147 // `Object(arg0, ...)`
1148 std::string sig = it2->signature;
1149 size_t start = sig.find('(') + 7; // skip "(self: "
1150 if (start < sig.size()) {
1151 // End at the , for the next argument
1152 size_t end = sig.find(", "), next = end + 2;
1153 size_t ret = sig.rfind(" -> ");
1154 // Or the ), if there is no comma:
1155 if (end >= sig.size()) {
1156 next = end = sig.find(')');
1157 }
1158 if (start < end && next < sig.size()) {
1159 msg.append(sig, start, end - start);
1160 msg += '(';
1161 msg.append(sig, next, ret - next);
1162 wrote_sig = true;
1163 }
1164 }
1165 }
1166 if (!wrote_sig) {
1167 msg += it2->signature;
1168 }
1169
1170 msg += '\n';
1171 }
1172 msg += "\nInvoked with: ";
1173 auto args_ = reinterpret_borrow<tuple>(args_in);
1174 bool some_args = false;
1175 for (size_t ti = overloads->is_constructor ? 1 : 0; ti < args_.size(); ++ti) {
1176 if (!some_args) {
1177 some_args = true;
1178 } else {
1179 msg += ", ";
1180 }
1181 try {
1182 msg += pybind11::repr(args_[ti]);
1183 } catch (const error_already_set &) {
1184 msg += "<repr raised Error>";
1185 }
1186 }
1187 if (kwargs_in) {
1188 auto kwargs = reinterpret_borrow<dict>(kwargs_in);
1189 if (!kwargs.empty()) {
1190 if (some_args) {
1191 msg += "; ";
1192 }
1193 msg += "kwargs: ";
1194 bool first = true;
1195 for (const auto &kwarg : kwargs) {
1196 if (first) {
1197 first = false;
1198 } else {
1199 msg += ", ";
1200 }
1201 msg += pybind11::str("{}=").format(kwarg.first);
1202 try {
1203 msg += pybind11::repr(kwarg.second);
1204 } catch (const error_already_set &) {
1205 msg += "<repr raised Error>";
1206 }
1207 }
1208 }
1209 }
1210
1211 append_note_if_missing_header_is_suspected(msg);
1212 // Attach additional error info to the exception if supported
1213 if (PyErr_Occurred()) {
1214 // #HelpAppreciated: unit test coverage for this branch.
1215 raise_from(PyExc_TypeError, msg.c_str());
1216 return nullptr;
1217 }
1218 set_error(PyExc_TypeError, msg.c_str());
1219 return nullptr;
1220 }
1221 if (!result) {
1222 std::string msg = "Unable to convert function return value to a "
1223 "Python type! The signature was\n\t";
1224 assert(current_overload != nullptr);
1225 msg += current_overload->signature;
1226 append_note_if_missing_header_is_suspected(msg);
1227 // Attach additional error info to the exception if supported
1228 if (PyErr_Occurred()) {
1229 raise_from(PyExc_TypeError, msg.c_str());
1230 return nullptr;
1231 }
1232 set_error(PyExc_TypeError, msg.c_str());
1233 return nullptr;
1234 }
1235 if (overloads->is_constructor && !self_value_and_holder.holder_constructed()) {
1236 auto *pi = reinterpret_cast<instance *>(parent.ptr());
1237 self_value_and_holder.type->init_instance(pi, nullptr);
1238 }
1239 return result.ptr();
1240 }

2919 if (!nurse || !patient) {
2920 pybind11_fail("Could not activate keep_alive!");
2921 }
2922
2923 if (patient.is_none() || nurse.is_none()) {
2924 return; /* Nothing to keep alive or nothing to be kept alive by */
2925 }

2927 auto tinfo = all_type_info(Py_TYPE(nurse.ptr()));
2928 if (!tinfo.empty()) {
2929 /* It's a pybind-registered type, so we can store the patient in the
2930 * internal list. */

2934 * Boost.Python. This is not used for pybind-registered types because

2936 cpp_function disable_lifesupport([patient](handle weakref) {
2937 patient.dec_ref();
2938 weakref.dec_ref();
2939 });

2941 weakref wr(nurse, disable_lifesupport);
2942
2943 patient.inc_ref(); /* reference patient and leak the weak reference */
2944 (void) wr.release();
2945 }
