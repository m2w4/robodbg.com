264#if defined(PYBIND11_TYPING_H_HAS_STRING_LITERAL)
265template <typing::StringLiteral StrLit>
266consteval auto sanitize_string_literal() {
267 constexpr std::string_view v(StrLit.name);
268 constexpr std::string_view special_chars("!@%{}-");
269 constexpr auto num_special_chars = std::accumulate(
270 special_chars.begin(), special_chars.end(), (size_t) 0, [&v](auto acc, const char &c) {
271 return std::move(acc) + std::ranges::count(v, c);
272 });
273 char result[v.size() + num_special_chars + 1];
274 size_t i = 0;
275 for (auto c : StrLit.name) {
276 if (special_chars.find(c) != std::string_view::npos) {
277 result[i++] = '!';
278 }
279 result[i++] = c;
280 }
281 return typing::StringLiteral(result);
282}
