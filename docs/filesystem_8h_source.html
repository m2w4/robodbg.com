33#if defined(PYBIND11_HAS_FILESYSTEM) || defined(PYBIND11_HAS_EXPERIMENTAL_FILESYSTEM)
34template <typename T>
35struct path_caster {
36
37private:
38 static PyObject *unicode_from_fs_native(const std::string &w) {
39# if !defined(PYPY_VERSION)
40 return PyUnicode_DecodeFSDefaultAndSize(w.c_str(), ssize_t(w.size()));
41# else
42 // PyPy mistakenly declares the first parameter as non-const.
43 return PyUnicode_DecodeFSDefaultAndSize(const_cast<char *>(w.c_str()), ssize_t(w.size()));
44# endif
45 }
46
47 static PyObject *unicode_from_fs_native(const std::wstring &w) {
48 return PyUnicode_FromWideChar(w.c_str(), ssize_t(w.size()));
49 }
50
51public:
52 static handle cast(const T &path, return_value_policy, handle) {
53 if (auto py_str = unicode_from_fs_native(path.native())) {
54 return module_::import("pathlib")
55 .attr("Path")(reinterpret_steal<object>(py_str))
56 .release();
57 }
58 return nullptr;
59 }
60
61 bool load(handle handle, bool) {
62 // PyUnicode_FSConverter and PyUnicode_FSDecoder normally take care of
63 // calling PyOS_FSPath themselves, but that's broken on PyPy (PyPy
64 // issue #3168) so we do it ourselves instead.
65 PyObject *buf = PyOS_FSPath(handle.ptr());
66 if (!buf) {
67 PyErr_Clear();
68 return false;
69 }
70 PyObject *native = nullptr;
71 if constexpr (std::is_same_v<typename T::value_type, char>) {
72 if (PyUnicode_FSConverter(buf, PYBIND11_REINTERPRET_CAST_VOID_PTR_IF_NOT_PYPY(&native))
73 != 0) {
74 if (auto *c_str = PyBytes_AsString(native)) {
75 // AsString returns a pointer to the internal buffer, which
76 // must not be free'd.
77 value = c_str;
78 }
79 }
80 } else if constexpr (std::is_same_v<typename T::value_type, wchar_t>) {
81 if (PyUnicode_FSDecoder(buf, PYBIND11_REINTERPRET_CAST_VOID_PTR_IF_NOT_PYPY(&native))
82 != 0) {
83 if (auto *c_str = PyUnicode_AsWideCharString(native, nullptr)) {
84 // AsWideCharString returns a new string that must be free'd.
85 value = c_str; // Copies the string.
86 PyMem_Free(c_str);
87 }
88 }
89 }
90 Py_XDECREF(native);
91 Py_DECREF(buf);
92 if (PyErr_Occurred()) {
93 PyErr_Clear();
94 return false;
95 }
96 return true;
97 }
98
99 PYBIND11_TYPE_CASTER(T, io_name("os.PathLike | str | bytes", "pathlib.Path"));
100};
