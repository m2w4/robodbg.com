13bool Debugger::hideDebugger( ) {
14 PROCESS_BASIC_INFORMATION pbi = {};
15 ULONG retLen;
16
17 NTSTATUS status = NtQueryInformationProcess(
18 hProcessGlobal,
19 ProcessBasicInformation,
20 &pbi,
21 sizeof(pbi),&retLen
22 );
23
24 if (status != 0) {
25 // std::cerr << "[-] NtQueryInformationProcess failed\n";
26 return false;
27 }
28
29 // PEB is at this address in remote process
30 BYTE beingDebugged = 0;
31 SIZE_T bytesWritten = 0;
32
33 // BeingDebugged is at offset 0x2 of the PEB
34 LPVOID pebDebugFlagAddr = (LPBYTE)pbi.PebBaseAddress + 2;
35
36 // Write zero to the flag
37 if (!WriteProcessMemory(hProcessGlobal, pebDebugFlagAddr, &beingDebugged, sizeof(beingDebugged), &bytesWritten)) {
38 std::cerr << "[-] WriteProcessMemory failed: " << GetLastError() << "\n";
39 return false;
40 }
41
42 return true;
43}

53// Attach to an existing process
54int Debugger::attachToProcess(std::string exeName) {
55 DWORD pid = Util::findProcessId(exeName);
56 if (pid == 0) {
57 std::cerr << "Process not found: " << exeName << std::endl;
58 return -1;
59 }
60 std::cout << "Attach to " << pid << std::endl;
61 if (!DebugActiveProcess(pid)) {
62 std::cerr << "Failed to attach to process: " << GetLastError() << std::endl;
63 return -1;
64 }
65
66 hProcessGlobal = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid);
67 if (hProcessGlobal == NULL) {
68 std::cerr << "Failed to open process: " << GetLastError() << std::endl;
69 return -1;
70 }
71 debuggedPid = pid;
72 onAttach( );
73 return 0;
74}

122void Debugger::decrementIP(HANDLE hThread) {
123 if (SuspendThread(hThread) == (DWORD)-1) {
124 std::cerr << "[-] SuspendThread failed: " << GetLastError() << "\n";
125 }
126 CONTEXT ctx = {};
127 ctx.ContextFlags = CONTEXT_CONTROL;
128 if (GetThreadContext(hThread, &ctx)) {
129 #ifdef _WIN64
130 ctx.Rip--;
131 #else
132 ctx.Eip--;
133 #endif
134 SetThreadContext(hThread, &ctx);
135 }
136 // Always try to resume
137 if (ResumeThread(hThread) == (DWORD)-1) {
138 std::cerr << "[-] ResumeThread failed: " << GetLastError() << "\n";
139 }
140}

156//actualize a thread list. e.g. after attaching to an existing running application;
157void Debugger::actualizeThreadList() {
158 threads.clear(); // Clear existing thread list
159
160 DWORD processId = GetProcessId(hProcessGlobal);
161 if (processId == 0) {
162 std::cerr << "Failed to get process ID from handle. Error: " << GetLastError() << std::endl;
163 return;
164 }
165
166 HANDLE snapshot = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
167 if (snapshot == INVALID_HANDLE_VALUE) {
168 std::cerr << "Failed to create thread snapshot. Error: " << GetLastError() << std::endl;
169 return;
170 }
171
172 THREADENTRY32 te32 = {};
173 te32.dwSize = sizeof(te32);
174
175 if (Thread32First(snapshot, &te32)) {
176 do {
177 if (te32.th32OwnerProcessID == processId) {
178 HANDLE hThread = OpenThread(THREAD_ALL_ACCESS, FALSE, te32.th32ThreadID);
179 if (!hThread) {
180 std::cerr << "Failed to open thread " << te32.th32ThreadID << ". Error: " << GetLastError() << std::endl;
181 continue;
182 }
183
184 // Optionally get thread start address (Windows internal APIs, see note below)
185 // For now we fill with nullptr placeholders
186 thread_t t;
187 t.hThread = hThread;
188 t.threadId = te32.th32ThreadID;
189 t.threadBase = nullptr; // Not easily available without NtQueryInformationThread
190 t.startAddress = nullptr; // Optional: can use NtQueryInformationThread with ThreadQuerySetWin32StartAddress
191
192 threads.push_back(t);
193 }
194 } while (Thread32Next(snapshot, &te32));
195 }
196
197 CloseHandle(snapshot);
198}

200int Debugger::loop() {
201 bool stepping = false;
202
203 DEBUG_EVENT dbgEvent;
204 while (true) {
205 if (!WaitForDebugEvent(&dbgEvent, INFINITE)) break;
206
207 DWORD cont = DBG_CONTINUE;
208
209 switch (dbgEvent.dwDebugEventCode) {
210 case CREATE_PROCESS_DEBUG_EVENT: {
211 imageBase = reinterpret_cast<uintptr_t>(dbgEvent.u.CreateProcessInfo.lpBaseOfImage);
212 DWORD_PTR entryPoint = Util::getEntryPoint(hProcessGlobal, reinterpret_cast<LPVOID>(imageBase));
213 onStart(imageBase,static_cast<uintptr_t>(entryPoint));
214 break;
215 }
216
217 case EXIT_PROCESS_DEBUG_EVENT: {
218 DWORD exitCode = dbgEvent.u.ExitProcess.dwExitCode;
219 DWORD pid = dbgEvent.dwProcessId;
220 onEnd(exitCode, pid);
221 return 0;
222 }
223 case CREATE_THREAD_DEBUG_EVENT: {
224 //std::cout << "[*] Thread created. TID=" << dbgEvent.dwThreadId
225 //<< " TEB=0x" << std::hex << (DWORD_PTR)threadBase
226 // << " Start Address=0x" << std::hex << (DWORD_PTR)threadStartAddr << "\n";
227
228 LPVOID threadBase = dbgEvent.u.CreateThread.lpThreadLocalBase;
229 LPVOID threadStartAddr = dbgEvent.u.CreateThread.lpStartAddress;
230
231 HANDLE hThread = OpenThread(THREAD_ALL_ACCESS, FALSE, dbgEvent.dwThreadId);
232 if (!hThread) {
233 std::cerr << "[-] Failed to open thread: " << GetLastError() << "\n";
234 break;
235 }
236 onThreadCreate( hThread, dbgEvent.dwThreadId, reinterpret_cast<uintptr_t>(threadBase), reinterpret_cast<uintptr_t>(threadStartAddr));
237
238 // Store the thread info
239 thread_t newThread = {
240 .hThread = hThread,
241 .threadId = dbgEvent.dwThreadId,
242 .threadBase = threadBase,
243 .startAddress = threadStartAddr
244 };
245 threads.push_back(newThread);
246
247 break;
248 }
249
250 case EXIT_THREAD_DEBUG_EVENT: {
251 //std::cout << "[*] Thread exited. TID=" << dbgEvent.dwThreadId << "\n";
252 onThreadExit( dbgEvent.dwThreadId );
253 break;
254 }
255
256 case LOAD_DLL_DEBUG_EVENT: {
257 LPVOID base = dbgEvent.u.LoadDll.lpBaseOfDll;
258 auto name = Util::getDllName(hProcessGlobal, dbgEvent.u.LoadDll.lpImageName, dbgEvent.u.LoadDll.fUnicode);
259 DWORD_PTR entryPoint = Util::getEntryPoint(hProcessGlobal, base);
260 onDLLLoad(reinterpret_cast<uintptr_t>(base), name, static_cast<uintptr_t>(entryPoint));
261
262 //std::cout << "[*] DLL loaded at 0x" << std::hex << (DWORD_PTR)base
263 //<< " Name: " << name << "\n";
264 break;
265 }
266
267 case UNLOAD_DLL_DEBUG_EVENT: {
268 LPVOID base = dbgEvent.u.UnloadDll.lpBaseOfDll;
269 auto name = Util::getDllName(hProcessGlobal, dbgEvent.u.LoadDll.lpImageName, dbgEvent.u.LoadDll.fUnicode);
270 onDLLUnload(reinterpret_cast<uintptr_t>(base), name);
271 //std::cout << "[*] DLL unloaded from 0x" << std::hex << (DWORD_PTR)base << "\n";
272 break;
273 }
274
275 case EXCEPTION_DEBUG_EVENT: {
276 DWORD code = dbgEvent.u.Exception.ExceptionRecord.ExceptionCode;
277 LPVOID addr = dbgEvent.u.Exception.ExceptionRecord.ExceptionAddress;
278 HANDLE hThread = OpenThread(THREAD_ALL_ACCESS, FALSE, dbgEvent.dwThreadId);
279 if (!hThread) break;
280
281 if (code == EXCEPTION_BREAKPOINT) {
282 if(this->verbose) std::cout << "[~] Breakpoint Hit" << std::endl;
283 auto it = breakpoints.find(addr);
284 if (it != breakpoints.end()) {
285 //std::cout << "[*] Loop Breakpoint hit at: 0x" << std::hex << (DWORD_PTR)addr << "\n";
286 restoreBreakpoint(addr);
287 decrementIP(hThread);
288 BreakpointAction bpType = onBreakpoint(reinterpret_cast<uintptr_t>(addr),hThread);
289 if (bpType != BREAK ) {
290 enableSingleStep(hThread);
291 stepping = true;
292 if(bpType == RESTORE )
293 {
294 bpAddrToRestore = addr;
295 lastBpType = bpType;
296 needToRestoreBreakpoint = true;
297 }
298 } else { //BP_DO_NOP
299 stepping = false;
300 }
301 }
302 } else if (code == EXCEPTION_SINGLE_STEP && stepping) {
303 if(needToRestoreBreakpoint) //just hit a breakpoint. Now restore.
304 {
305 if(restoreHwBP)
306 {
307 if(this->verbose) std::cout << "[~] Restoring hardware breakpoint" << std::endl;
308 setHardwareBreakpoint(hwBPToRestore);
309 restoreHwBP = false;
310 } else {
311 setBreakpoint(bpAddrToRestore); //TODO
312 }
313 needToRestoreBreakpoint = false;
314
315 }
316
317 if(lastBpType == SINGLE_STEP ) {
318 BreakpointAction bpType = onBreakpoint(reinterpret_cast<uintptr_t>(addr),hThread);
319 if (bpType != BREAK) {
320 bpAddrToRestore = addr;
321 needToRestoreBreakpoint = true;
322
323 lastBpType = bpType;
324 enableSingleStep(hThread);
325 stepping = true;
326 }
327 } else {
328 stepping = false;
329 }
330 } else if (code == EXCEPTION_SINGLE_STEP ) {
331 if(this->verbose) std::cout << "[~] Single step: 0x" << std::hex << (DWORD_PTR)addr << "\n";
332 DRReg reg = isHardwareBreakpointAt( addr );
333 if(reg != DRReg::NOP) {
334 BreakpointAction bpType = onHardwareBreakpoint(reinterpret_cast<uintptr_t>(addr), hThread, reg);
335 if(bpType == BREAK) { //Software Breakpoint
336 clearHardwareBreakpoint( reg );
337 } else if(bpType == SINGLE_STEP)
338 {
339 enableSingleStep(hThread);
340 stepping = true;
341 clearHardwareBreakpoint(reg);
342 } else if(bpType == RESTORE)
343 {
344 hwBp_t bp = getBreakpointByReg( reg );
345 needToRestoreBreakpoint = true;
346 restoreHwBP = true;
347 hwBPToRestore = bp;
348 bpAddrToRestore = addr;
349 hwBPThreadToRestore = hThread;
350 lastBpType = bpType;
351
352 //getHardwareBreakpoints();
353 clearHardwareBreakpointOnThread(hwBPThreadToRestore, reg);
354 enableSingleStep(hThread);
355 stepping = true;
356 }
357 } else {
358 onUnknownException(reinterpret_cast<uintptr_t>(addr), code);
359
360 }
361 } else if ( code == EXCEPTION_ACCESS_VIOLATION ) {
362 ULONG_PTR rawAddr = dbgEvent.u.Exception.ExceptionRecord.ExceptionInformation[1];
363 LPVOID faultingAddress = reinterpret_cast<LPVOID>(rawAddr);
364 ULONG_PTR accessType = dbgEvent.u.Exception.ExceptionRecord.ExceptionInformation[0];
365 //MemoryRegion_t mr = getPageByAddress( faultingAddress);
366 //changeMemoryProtection(mr, PAGE_EXECUTE_READWRITE);
367 onAccessViolation( static_cast<uintptr_t>(rawAddr), reinterpret_cast<uintptr_t>(faultingAddress), static_cast<long>(accessType));
368 //enableSingleStep(hThread);
369 //stepping = true;
370
371 } else {
372 onUnknownException( reinterpret_cast<uintptr_t>(addr), code );
373 }
374
375 CloseHandle(hThread);
376 break;
377 }
378
379 case OUTPUT_DEBUG_STRING_EVENT: {
380 OUTPUT_DEBUG_STRING_INFO& info = dbgEvent.u.DebugString;
381 SIZE_T bytesRead;
382
383 if (!info.fUnicode) {
384 // ANSI string
385 char buffer[1024] = {0};
386 ReadProcessMemory(
387 hProcessGlobal,
388 info.lpDebugStringData,
389 buffer,
390 std::min<DWORD>(info.nDebugStringLength, sizeof(buffer) - 1),
391 &bytesRead
392 );
393
394 std::string msg(buffer);
395 onDebugString(msg);
396 } else {
397 // Unicode string
398 wchar_t wbuffer[1024] = {0};
399 ReadProcessMemory(
400 hProcessGlobal,
401 info.lpDebugStringData,
402 wbuffer,
403 std::min<DWORD>(info.nDebugStringLength * sizeof(wchar_t), sizeof(wbuffer) - sizeof(wchar_t)),
404 &bytesRead
405 );
406
407 // Convert wchar_t* to std::string using WideCharToMultiByte (UTF-8)
408 int size_needed = WideCharToMultiByte(CP_UTF8, 0, wbuffer, -1, nullptr, 0, nullptr, nullptr);
409 std::string msg(size_needed, 0);
410 WideCharToMultiByte(CP_UTF8, 0, wbuffer, -1, &msg[0], size_needed, nullptr, nullptr);
411
412 onDebugString(msg);
413 }
414
415 break;
416 }
417
418 case RIP_EVENT: {
419 const RIP_INFO& rip = dbgEvent.u.RipInfo;
420 onRIPError( rip );
421 break;
422 }
423
424 default: {
425 onUnknownDebugEvent( dbgEvent.dwDebugEventCode );
426 break;
427 }
428 }
429 ContinueDebugEvent(dbgEvent.dwProcessId, dbgEvent.dwThreadId, cont);
430 }
431
432 CloseHandle(hProcessGlobal);
433 CloseHandle(hThreadGlobal);
434 return 0;
435}
